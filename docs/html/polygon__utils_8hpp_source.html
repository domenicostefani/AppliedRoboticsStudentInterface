53  bool pointOrderComparison(Point a, Point b, Point center)
54  {
55  if (a.x - center.x >= 0 && b.x - center.x < 0)
56  return true;
57  if (a.x - center.x < 0 && b.x - center.x >= 0)
58  return false;
59  if (a.x - center.x == 0 && b.x - center.x == 0) {
60  if (a.y - center.y >= 0 || b.y - center.y >= 0)
61  return a.y > b.y;
62  return b.y > a.y;
63  }
64 
65  // compute the cross product of vectors (center -> a) x (center -> b)
66  float det = (a.x - center.x) * (b.y - center.y) - (b.x - center.x) * (a.y - center.y);
67  if (det < 0)
68  return true;
69  if (det > 0)
70  return false;
71 
72  // points a and b are on the same line from the center
73  // check which point is closer to the center
74  float d1 = (a.x - center.x) * (a.x - center.x) + (a.y - center.y) * (a.y - center.y);
75  float d2 = (b.x - center.x) * (b.x - center.x) + (b.y - center.y) * (b.y - center.y);
76  return d1 > d2;
77  }

116  float cAB = pA.y - (mAB * pA.x);
117  // Find line perpendicular line to the one for A and B, passing for P
