151 cv::Point findLineCenter(const cv::Mat& img_in, const std::vector<cv::Point> &arena) {
152  cv::Mat lower_red_hue_range; // the lower range for red hue
153  cv::Mat upper_red_hue_range; // the higher range for red hue
154  cv::Mat hsv_img;
155  cv::cvtColor(img_in, hsv_img, cv::COLOR_BGR2HSV);
156  cv::inRange(hsv_img, cv::Scalar(0, 100, 100), cv::Scalar(10, 255, 255), lower_red_hue_range);
157  cv::inRange(hsv_img, cv::Scalar(160, 100, 100), cv::Scalar(180, 255, 255), upper_red_hue_range);
158  cv::Mat red_hue_image;
159  cv::addWeighted(lower_red_hue_range, 1.0, upper_red_hue_range, 1.0, 0.0, red_hue_image);
160 
161  cv::Size size(red_hue_image.cols, red_hue_image.rows);
162  cv::Mat maskRect(size, CV_8U, cv::Scalar(0.0));
163 
164  std::vector<cv::Point2f> tempcorners;
165  for (int i = 0; i < 4; ++i)
166  tempcorners.emplace_back(arena[i].x, arena[i].y);
167  cv::Scalar white = cv::Scalar(255.0); // white
168  cv::Point vertices[4];
169  for (int i = 0; i < 4; ++i) {
170  vertices[i] = tempcorners[i];
171  }
172  cv::fillConvexPoly(maskRect, vertices, 4, white);
173 
174  // compute mask
175  cv::Mat maskContours;
176  cv::threshold(red_hue_image, maskContours, 120, 255, CV_THRESH_BINARY_INV | CV_THRESH_OTSU);
177  cv::subtract(red_hue_image, maskRect, maskContours);
178 
179  std::vector<std::vector<cv::Point>> contoursForLine;
180  cv::findContours(maskContours,contoursForLine, CV_RETR_EXTERNAL, CV_CHAIN_APPROX_SIMPLE);
181  // Find biggest contour (if there are other contours in the image, we assume
182  // the biggest one is the desired rect)
183  int biggestContourIdxForLine = -1;
184  float biggestContourAreaForLine = 0;
185 
186  for (size_t i = 0; i< contoursForLine.size(); i++) {
187  float ctArea= cv::contourArea(contoursForLine[i]);
188  if(ctArea > biggestContourAreaForLine)
189  {
190  biggestContourAreaForLine = ctArea;
191  biggestContourIdxForLine = i;
192  }
193  }
194  std::vector<cv::Point> approx_curve_forLine;
195  approxPolyDP(contoursForLine[biggestContourIdxForLine], approx_curve_forLine,4, true);
196 
197  int xsum = 0;
198  int ysum = 0;
199  for (int i = 0; i < 4; i++) {
200  xsum += approx_curve_forLine[i].x;
201  ysum += approx_curve_forLine[i].y;
202  }
