98 Polygon inflateWithClipper(const Polygon& poly, float amount) {
99  double delta = amount * clipperScale; // scale the inflation amount
100 
101  ClipperLib::Path toInflate = Polygon2CPath(poly); // convert to Clipper path
102  ClipperLib::Paths inflated;
103  ClipperLib::ClipperOffset co;
104  co.AddPath(toInflate, ClipperLib::jtSquare, ClipperLib::etClosedPolygon);
105  co.Execute(inflated, delta);
106  assert(inflated.size() == 1);
107 
108  return CPath2Polygon(inflated[0]);
109 }

138 Polygon offsetBorders(const Polygon& borders, float amount) {
139  Polygon correctedBorders = ClipperHelper::inflateWithClipper(borders, amount);
140  assert(borders.size() == correctedBorders.size());
141 
142  // Reorder points
143  {
144  // Find first point
145  int closerToFirst = -1;
146  float minDistance = std::numeric_limits<float>::max();
147  Polygon correctedBordersCopy = correctedBorders;
148  for (size_t i = 0; i < correctedBorders.size(); ++i) {
149  float distanceFromFirst = pow(borders[0].x - correctedBordersCopy[i].x,2) + pow(borders[0].y - correctedBordersCopy[i].y,2);
150  if (distanceFromFirst < minDistance) {
151  minDistance = distanceFromFirst;
152  closerToFirst = i;
153  }
154  }
155 
156  assert(closerToFirst != -1);
157 
158  // Actual reorder
159  for (size_t i = 0; i < correctedBorders.size(); ++i) {
160  int copyIndex = (i + closerToFirst)%correctedBorders.size();
161  correctedBorders[i] = correctedBordersCopy[copyIndex];
162  }
163  }
164 
165  return correctedBorders;
166 }
