145 
146  // Reorder points
147  {

149  int closerToFirst = -1;

151  Polygon correctedBordersCopy = correctedBorders;
152  for (size_t i = 0; i < correctedBorders.size(); ++i) {
153  float distanceFromFirst = pow(borders[0].x - correctedBordersCopy[i].x,2) + pow(borders[0].y - correctedBordersCopy[i].y,2);
154  if (distanceFromFirst < minDistance) {
